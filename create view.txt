CREATE VIEW BOOKINFO
AS 
(SELECT     BOOK.BOOK_ID AS ID, COVER, ISBN, PRICE, PUBLICATION_YEAR, CATEGORY, PUBLISHER, 
                      TITLE, RANKING, HARD_COVER, AVAILABILITY, DESCRIPTION_FIELD, AUTHOR, PAGES, 
                      RATE, COMMENTS, NICKNAME
FROM         BOOK, REVIEW, CUSTOMER
WHERE BOOK.BOOK_ID = REVIEW.BOOK_ID  AND REVIEW.CUSTOMER_ID = CUSTOMER.CUSTOMER_ID);
